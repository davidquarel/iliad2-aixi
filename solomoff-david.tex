\mytitle{Solomonoff Worksheet --- Pareto Optimality \hfill \today}

\medskip
\noindent
This is a self-contained companion to the main Solomonoff worksheet.
It establishes that the Bayesian mixture $\xi$ is \emph{uniquely Pareto-optimal}
under both squared and KL prediction loss --- no other predictor can do at
least as well as $\xi$ on every environment in the class without coinciding
with $\xi$ entirely.

The shape of both proofs is identical: state a one-line ``Pythagorean''
decomposition of the $w$-weighted loss around $\xi$, then read off uniqueness
from non-negativity. This decomposition is the key idea: it turns both
Pareto-optimality proofs into two-liners and, as a bonus, shows that $\xi$ is
actually the \emph{minimizer} of the $w$-weighted loss across all
predictors --- a much stronger statement than Pareto optimality alone.

Difficulty ratings follow Knuth's scheme: \textbf{[05]} a few minutes,
\textbf{[10]} 15-minute pencil-and-paper, \textbf{[15]} half an hour or so.

%===================================
\section*{Recap of setup}

We use the same setup as the main worksheet: countable model class
$\Msol = \{\nu_1, \nu_2, \dots\}$ of environments (each $\nu$ assigning a
proper distribution $\nu(\cdot \mid x_{<t}) \in \Delta\mathbb{B}$ to every
history), prior $w_\nu > 0$ with
$\sum_\nu w_\nu = 1$, and Bayesian mixture
\[
  \xi(x) ~:=~ \sum_{\nu \in \Msol} w_\nu \, \nu(x),
  \qquad
  \xi(x_t \mid x_{<t}) ~:=~ \sum_{\nu \in \Msol} w(\nu \mid x_{<t}) \, \nu(x_t \mid x_{<t}).
\]
A \textbf{predictor} is any function $\rho$ that assigns to each history
$x_{<t}$ a distribution $\rho(\cdot \mid x_{<t}) \in \Delta\mathbb{B}$; by
chain rule, $\rho(x_{1:n}) := \prod_t \rho(x_t \mid x_{<t})$. The mixture
$\xi$ is one such predictor, and any $\nu \in \Msol$ is another.

\begin{definition}[Two prediction losses]\label{def:losses}
Fix a horizon $n \geq 1$. For environments $\nu$ and predictors $\rho$, define
\begin{align*}
  \FSQ^n(\nu, \rho)
    ~&:=~
    \sum_{t=1}^{n}\;
    \sum_{x_{<t} \in \mathbb{B}^{t-1}}\;
    \sum_{x_t \in \mathbb{B}}\;
    \bigl(\nu(x_t \mid x_{<t}) - \rho(x_t \mid x_{<t})\bigr)^{2}, \\[4pt]
  \FKL^n(\nu, \rho)
    ~&:=~
    \KL\bigl(\nu(X_{1:n}) \,\big\|\, \rho(X_{1:n})\bigr)
    ~=~
    \sum_{x_{1:n} \in \mathbb{B}^n}
      \nu(x_{1:n}) \, \ln \frac{\nu(x_{1:n})}{\rho(x_{1:n})}.
\end{align*}
$\FSQ^n$ is the cumulative pointwise squared error of conditional predictions
up to horizon $n$, summed over all histories (no $\mu$-weighting --- it is a
purely geometric distance between predictors). $\FKL^n$ is the joint KL of
$\nu$ and $\rho$ at horizon $n$. We will fix $n$ throughout; statements below
all hold for every $n$.
\end{definition}

\begin{definition}[Pareto domination]\label{def:pareto}
A predictor $\rho$ \textbf{weakly Pareto-dominates} $\xi$ under loss
$\mathcal F$ if
\[
  \mathcal F(\nu, \rho) ~\leq~ \mathcal F(\nu, \xi)
  \qquad \text{for every } \nu \in \Msol.
\]
$\xi$ is \textbf{Pareto-optimal} if the only predictor that weakly dominates
it is $\xi$ itself.\footnote{This is stronger than the usual ``no strict
domination'' formulation: weak domination already forces $\rho = \xi$, so a
fortiori there is no $\rho$ that is strictly better on some $\nu$ without being
worse on another.}
\end{definition}

%===================================
\section{Two Pythagorean identities for the mixture}\label{prob:pythag}

The next lemma is the engine of everything that follows. In both parts the
mechanism is identical: $\xi$ is the $w$-weighted mean of the
$\nu$'s, so adding and subtracting $\xi$ separates the loss into
``how spread out the $\nu$'s are around $\xi$'' plus ``how far $\rho$ is from
$\xi$.'' The cross-term vanishes because $\sum_\nu w_\nu (\nu - \xi) = 0$.

\bigskip

\noindent
\textbf{[10]} (Squared-loss decomposition.) For every history $x_{<t}$, symbol
$x_t \in \mathbb{B}$, and every predictor $\rho$, show
\begin{align*}
  &\bigl(\xi(x_t \mid x_{<t}) - \rho(x_t \mid x_{<t})\bigr)^{2}
  ~+~
  \sum_{\nu \in \Msol} w_\nu \bigl(\nu(x_t \mid x_{<t}) - \xi(x_t \mid x_{<t})\bigr)^{2} \\
  &=~
  \sum_{\nu \in \Msol} w_\nu \bigl(\nu(x_t \mid x_{<t}) - \rho(x_t \mid x_{<t})\bigr)^{2}.
\end{align*}

\textit{Hint.} Add and subtract $\xi(x_t \mid x_{<t})$ inside the squared term,
then expand. Use $\sum_\nu w_\nu = 1$ and $\sum_\nu w_\nu \nu(x_t \mid x_{<t})
= \xi(x_t \mid x_{<t})$ to kill the cross-term.

\begin{solution}
Abbreviate $\nu := \nu(x_t \mid x_{<t})$, $\xi := \xi(x_t \mid x_{<t})$,
$\rho := \rho(x_t \mid x_{<t})$ for the duration of the calculation.

\medskip
\textit{Step 1: $\xi$ is the $w$-weighted mean of the $\nu$'s.}
By definition of the mixture and the prior normalization $\sum_\nu w_\nu = 1$,
\[
  \sum_\nu w_\nu \, \nu ~=~ \xi.
\]

\medskip
\textit{Step 2: the cross-term vanishes.}
A direct consequence of Step 1, again using $\sum_\nu w_\nu = 1$:
\[
  \sum_\nu w_\nu \, (\nu - \xi)
  ~=~ \sum_\nu w_\nu \, \nu \;-\; \xi \sum_\nu w_\nu
  ~=~ \xi - \xi ~=~ 0.
\]

\medskip
\textit{Step 3: expand the square.}
Write $\nu - \rho = (\nu - \xi) + (\xi - \rho)$ and expand:
\begin{align*}
\sum_\nu w_\nu (\nu - \rho)^2
  &= \sum_\nu w_\nu\bigl[(\nu - \xi)^2 + 2(\nu - \xi)(\xi - \rho) + (\xi - \rho)^2\bigr] \\
  &= \sum_\nu w_\nu (\nu - \xi)^2
     \;+\; 2(\xi - \rho)\sum_\nu w_\nu (\nu - \xi)
     \;+\; (\xi - \rho)^2 \sum_\nu w_\nu.
\end{align*}
The middle sum is zero by Step 2; the trailing $\sum_\nu w_\nu = 1$. So
\[
  (\xi - \rho)^2 \;+\; \sum_\nu w_\nu (\nu - \xi)^2
  ~=~ \sum_\nu w_\nu (\nu - \rho)^2. \qedhere
\]
\end{solution}

\bigskip

\noindent
\textbf{[10]} (KL decomposition / Bregman--Pythagoras.) Let $\rho$ be any
predictor with $\rho(x_{1:n}) > 0$ for all $x_{1:n}$. Show
\begin{align*}
  &\KL\bigl(\xi(X_{1:n}) \,\big\|\, \rho(X_{1:n})\bigr)
  ~+~
  \sum_{\nu \in \Msol} w_\nu \, \KL\bigl(\nu(X_{1:n}) \,\big\|\, \xi(X_{1:n})\bigr) \\
  &=~
  \sum_{\nu \in \Msol} w_\nu \, \KL\bigl(\nu(X_{1:n}) \,\big\|\, \rho(X_{1:n})\bigr).
\end{align*}

\textit{Hint.} Subtract the second sum from the third, simplify the
log-difference $\ln\tfrac{\nu}{\rho} - \ln\tfrac{\nu}{\xi} = \ln\tfrac{\xi}{\rho}$
inside the $\nu$-expectation, then swap the order of summation and use
$\sum_\nu w_\nu \nu(x_{1:n}) = \xi(x_{1:n})$.

\begin{solution}
Take the difference $\sum_\nu w_\nu \KL(\nu \,\|\, \rho) - \sum_\nu w_\nu
\KL(\nu \,\|\, \xi)$. Inside each KL the $\nu(x_{1:n}) \ln \nu(x_{1:n})$ pieces
cancel, leaving
\begin{align*}
\sum_\nu w_\nu \KL(\nu \,\|\, \rho) - \sum_\nu w_\nu \KL(\nu \,\|\, \xi)
&= \sum_\nu w_\nu \sum_{x_{1:n}}
   \nu(x_{1:n})\,\Bigl[\ln\tfrac{\nu(x_{1:n})}{\rho(x_{1:n})}
                  - \ln\tfrac{\nu(x_{1:n})}{\xi(x_{1:n})}\Bigr] \\
&= \sum_\nu w_\nu \sum_{x_{1:n}}
   \nu(x_{1:n})\,\ln \tfrac{\xi(x_{1:n})}{\rho(x_{1:n})}.
\end{align*}
The factor $\ln(\xi/\rho)$ does not depend on $\nu$, so we may swap the order
of summation and pull it out, then use $\sum_\nu w_\nu \nu(x_{1:n}) =
\xi(x_{1:n})$:
\begin{align*}
\sum_\nu w_\nu \sum_{x_{1:n}}
   \nu(x_{1:n})\,\ln \tfrac{\xi(x_{1:n})}{\rho(x_{1:n})}
&= \sum_{x_{1:n}} \ln \tfrac{\xi(x_{1:n})}{\rho(x_{1:n})}
   \underbrace{\sum_\nu w_\nu \nu(x_{1:n})}_{=\, \xi(x_{1:n})} \\
&= \sum_{x_{1:n}} \xi(x_{1:n}) \,\ln \tfrac{\xi(x_{1:n})}{\rho(x_{1:n})}
~=~ \KL\bigl(\xi(X_{1:n}) \,\big\|\, \rho(X_{1:n})\bigr).
\end{align*}
Rearranging gives the claimed identity. \qedhere
\end{solution}

\begin{remark}[Why these are ``Pythagorean'']
Both identities have the shape
\[
  \text{(loss from $\rho$, averaged over $\nu$'s)}
  \;=\;
  \text{(loss from $\xi$, averaged over $\nu$'s)} \;+\;
  \text{(loss from $\rho$ to $\xi$).}
\]
For squared loss this is exactly Pythagoras' theorem in Euclidean space, with
$\xi$ playing the role of the orthogonal projection of the ``cloud of $\nu$'s''
onto the line of possible reference predictors. For KL it is the Bregman-divergence
analogue: $\KL$ behaves like a squared distance, with $\xi$ as the Bregman
projection. The crucial property both proofs share is the
``mean-zero cross-term,'' which here is just $\sum_\nu w_\nu (\nu - \xi) = 0$.
\end{remark}

\begin{remark}[$\xi$ is the unique loss-minimizer]
Both identities show more than Pareto optimality. Read either identity as a
function of $\rho$: the $\rho$-dependence on the RHS is a single non-negative
term that vanishes only at $\rho = \xi$. So the unique minimizer of the
$w$-weighted loss
\[
  \rho \;\longmapsto\; \sum_\nu w_\nu \,\mathcal F(\nu, \rho)
\]
is $\rho = \xi$, for either choice of loss. The Pareto-optimality result of
\cref{prob:pareto} below is an immediate corollary.
\end{remark}

%===================================
\section{The Bayesian mixture is uniquely Pareto-optimal}\label{prob:pareto}

\noindent
\textbf{[05]} Using the identities of \cref{prob:pythag}, show that $\xi$ is
Pareto-optimal under both losses in the sense of \cref{def:pareto}: for
$\mathcal F \in \{\FSQ^n, \FKL^n\}$, if $\rho$ is any predictor with
\[
  \mathcal F(\nu, \rho) ~\leq~ \mathcal F(\nu, \xi)
  \qquad \text{for every } \nu \in \Msol,
\]
then $\rho(x_t \mid x_{<t}) = \xi(x_t \mid x_{<t})$ for every history $x_{<t}$
and every $x_t$.

\textit{Hint.} Take the $w$-weighted sum of the assumed inequality over
$\nu \in \Msol$, then apply the corresponding identity from
\cref{prob:pythag} to recognize the extra non-negative term --- which must vanish.

\begin{solution}
Suppose $\mathcal F(\nu, \rho) \leq \mathcal F(\nu, \xi)$ for every
$\nu \in \Msol$. Multiplying by $w_\nu \geq 0$ and summing,
\begin{equation}\label{eq:weighted_dom}
  \sum_\nu w_\nu\, \mathcal F(\nu, \rho)
  ~\leq~
  \sum_\nu w_\nu\, \mathcal F(\nu, \xi).
\end{equation}

\textbf{Squared loss.} The cumulative loss $\FSQ^n$ is a finite sum of
non-negative per-step terms, so we may swap the order of summation freely.
Applying the squared-loss identity of \cref{prob:pythag} pointwise at each
$(t, x_{<t}, x_t)$ and summing, write
\[
  \|\xi - \rho\|^2
  ~:=~
  \sum_{t=1}^n \sum_{x_{<t}} \sum_{x_t}
    \bigl(\xi(x_t \mid x_{<t}) - \rho(x_t \mid x_{<t})\bigr)^2
  ~\geq~ 0;
\]
then
\[
  \|\xi - \rho\|^2 \;+\; \sum_\nu w_\nu \, \FSQ^n(\nu, \xi)
  ~=~
  \sum_\nu w_\nu \, \FSQ^n(\nu, \rho).
\]
Combining with \cref{eq:weighted_dom},
\[
  \|\xi - \rho\|^2 \;+\; \sum_\nu w_\nu \,\FSQ^n(\nu, \xi)
  ~\leq~ \sum_\nu w_\nu \,\FSQ^n(\nu, \xi)
  \;\;\Longrightarrow\;\;
  \|\xi - \rho\|^2 ~\leq~ 0.
\]
Since $\|\xi - \rho\|^2 \geq 0$, every term vanishes: $\rho(x_t \mid x_{<t}) =
\xi(x_t \mid x_{<t})$ for every $(t, x_{<t}, x_t)$ with $t \leq n$.

\textbf{KL loss.} Directly from the KL identity of \cref{prob:pythag} and
\cref{eq:weighted_dom},
\[
  \KL\bigl(\xi(X_{1:n}) \,\|\, \rho(X_{1:n})\bigr) \;+\;
  \sum_\nu w_\nu \,\FKL^n(\nu, \xi)
  ~\leq~
  \sum_\nu w_\nu \,\FKL^n(\nu, \xi),
\]
so $\KL(\xi(X_{1:n}) \,\|\, \rho(X_{1:n})) \leq 0$. KL is non-negative, hence
$\KL(\xi(X_{1:n}) \,\|\, \rho(X_{1:n})) = 0$, which forces
$\xi(X_{1:n}) = \rho(X_{1:n})$ as joint distributions. By chain rule, the
conditionals agree as well. \qedhere
\end{solution}

\begin{remark}[Versus the standard ``Bayes is admissible'' result]
Pareto optimality is a kind of admissibility statement: nothing weakly
dominates $\xi$ across the class. The standard textbook way to prove this for
Bayesian estimators goes via the complete-class theorem in decision theory and
is considerably more abstract. The Pythagorean identity used here gives the
result by direct calculation, with no decision-theoretic machinery and no
assumptions beyond a proper prior ($\sum_\nu w_\nu = 1$). The reason it works
so cleanly is that both losses are convex Bregman divergences in their second
argument --- exactly the setting in which the Bayesian mixture is the unique
minimizer.
\end{remark}
